Radio

This component is used inside other components, such as RadioGroup and RadioOption. Please do not use it directly.

() => {
  const [check, setCheck] = React.useState(false);
  return (
    <Radio
      label="label"
      name="name"
      id="id"
      checked={check}
      disabled={false}
      onChange={() => setCheck(!check)}
    />
  );
};
